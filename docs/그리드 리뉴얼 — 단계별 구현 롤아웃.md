## §3. Stage GR-2 — 이동 + on_enter 트리거

### 작업 범위

**`run_manager.gd::try_move`** 본격 구현:
```gdscript
func try_move(dir: Vector2i) -> bool:
    if run_data.get("phase") == "event": return false
    var target: Vector2i = run_data["player_pos"] + dir
    if not _is_passable(target): return false
    if run_data["actions_remaining"] < 1: return false
    
    run_data["player_pos"] = target
    run_data["actions_remaining"] -= 1
    run_data["visited_tiles"][target] = true
    
    _check_on_enter(target)
    
    if run_data["actions_remaining"] <= 0:
        rest()
    
    state_changed.emit()
    return true


func _check_on_enter(pos: Vector2i) -> void:
    var slot = _get_on_enter_slot(pos)
    if slot.is_empty(): return
    if not _slot_passes_filters(slot): return
    
    seen_this_run[slot.id] = true
    match slot.kind:
        "event": _begin_event_phase(slot.event_id)
        "research": _enter_research()
        # combat / 그 외는 GR-5 에서
```

**`game_data.gd::TILE_ENCOUNTERS`** 채움:
```gdscript
const TILE_ENCOUNTERS = {
    Vector2i(8, 6): {
        "on_enter": {
            "id": "intro_at_spawn",
            "kind": "event",
            "event_id": "intro_speech",
            "once_per": "internal_run"
        }
    },
    Vector2i(12, 4): {
        "on_enter": {
            "id": "research_node",
            "kind": "research",
        }
    }
}
```

### 검증 — `test/test_grid_2.gd`

- `try_move(Vector2i(1, 0))` — 통과 가능 → 이동, actions=7
- `try_move(Vector2i(-1, -1))` — 벽 방향 → false, actions=7
- 이동 후 `visited_tiles[new_pos] = true`
- on_enter 슬롯 있는 타일 진입 → event 발화 (phase="event")
- once_per: internal_run 검증 — 같은 회차 재진입 시 미발화

## §4. Stage GR-3 — 탐험 액션 + explore_slot

### 작업 범위

```gdscript
func try_explore() -> bool:
    if run_data.get("phase") == "event": return false
    if run_data["actions_remaining"] < 1: return false
    
    var pos: Vector2i = run_data["player_pos"]
    if run_data["explored_tiles"].get(pos, false): return false
    
    run_data["explored_tiles"][pos] = true
    run_data["actions_remaining"] -= 1
    
    var slot = _get_explore_slot(pos)
    if not slot.is_empty() and _slot_passes_filters(slot):
        seen_this_run[slot.id] = true
        match slot.kind:
            "event": _begin_event_phase(slot.event_id)
    
    if run_data["actions_remaining"] <= 0:
        rest()
    
    state_changed.emit()
    return true
```

`TILE_ENCOUNTERS` 에 explore 슬롯 추가:
```gdscript
Vector2i(7, 5): {
    "explore": {
        "id": "find_letter_at_7_5",
        "kind": "event",
        "event_id": "find_letter",
        "once_per": "big_run"
    }
}
```

### 검증
- `try_explore()` 빈 타일 — false
- explore 슬롯 있는 타일 — event 발화
- 같은 회차 재탐험 — false
- 회귀 후 same big_run — once_per 따라 결정

## §5. Stage GR-4 — 휴식 / 일자 흐름

### 작업 범위

```gdscript
func rest() -> void:
    if run_data.get("phase") == "event": return
    
    # REST_EVENTS 추첨
    var rest_event_id = EventManager.resolve_event("on_rest", {})
    if rest_event_id != "":
        _begin_event_phase(rest_event_id)
        # event 종료 시 _on_event_resolved 가 phase 복귀
        # 그 핸들러가 rest 후처리도 같이 처리하도록 분기
    else:
        _finalize_rest()


func _finalize_rest() -> void:
    run_data["day"] += 1
    run_data["actions_remaining"] = ACTIONS_PER_DAY
    run_data["explored_tiles"] = {}   # 일자 reset? 정책 결정 — 회차 1 회로 갈지
    state_changed.emit()
```

**(주의)** 탐험 reset 정책 — §1 결정상 "한 회차 1 회" — 일자 단위로 reset 안 함. `explored_tiles` 는 회차 끝까지 영속, end_internal_run 시 리셋.

`game_data.gd::EVENTS` 에 REST_EVENTS 풀 추가:
```gdscript
"rest_dream_1": {
    "id": "rest_dream_1",
    "kind": "dialogue",
    "trigger": {"type": "on_rest"},
    "weight": 5,
    "lines": [{"speaker": "히로인", "text": "밤이 — 짧다."}]
}
```

### 검증
- actions=0 도달 시 자동 rest 호출
- day +1, actions 리셋
- on_rest 트리거된 이벤트 발화
- explored_tiles 유지 (회차 1 회 정책)

체크 ~10 항목.

## §7. Stage GR-6 — UI 그리드 시각 + 입력

### 작업 범위

**`run_ui.gd`**:
- `_build_map_display` 폐기 → `_build_grid_display`
- `_refresh_map_display` 폐기 → `_refresh_grid_display`
- `_node_buttons` 폐기 → `_tile_rects` 또는 별 결의 시각

```gdscript
const TILE_PX = 40
const GRID_ORIGIN = Vector2(180, 90)


func _build_grid_display():
    _grid_root = Control.new()
    _grid_root.position = GRID_ORIGIN
    _grid_root.visible = false
    add_child(_grid_root)
    
    var rows: int = GameData.WORLD_TERRAIN.size()
    var cols: int = GameData.WORLD_TERRAIN[0].length()
    for y in range(rows):
        for x in range(cols):
            var t: String = GameData.WORLD_TERRAIN[y][x]
            var rect = ColorRect.new()
            rect.position = Vector2(x * TILE_PX, y * TILE_PX)
            rect.size = Vector2(TILE_PX - 2, TILE_PX - 2)
            rect.color = _terrain_color(t)
            _grid_root.add_child(rect)
            _tile_rects[Vector2i(x, y)] = rect
    
    # 플레이어 마커
    _player_marker = ColorRect.new()
    _player_marker.size = Vector2(TILE_PX - 6, TILE_PX - 6)
    _player_marker.color = Color(1.0, 0.85, 0.3)
    _grid_root.add_child(_player_marker)


func _refresh_grid_display():
    var pos: Vector2i = RunManager.run_data["player_pos"]
    _player_marker.position = Vector2(pos.x * TILE_PX + 3, pos.y * TILE_PX + 3)
    
    for tile_pos in _tile_rects.keys():
        var rect: ColorRect = _tile_rects[tile_pos]
        var visited: bool = RunManager.run_data["visited_tiles"].get(tile_pos, false)
        rect.modulate = Color(1, 1, 1) if visited else Color(0.5, 0.5, 0.5)


func _unhandled_input(event: InputEvent) -> void:
    if RunManager.run_data.get("phase") != "map": return
    if not (event is InputEventKey) or not event.pressed: return
    match event.keycode:
        KEY_W, KEY_UP:    RunManager.try_move(Vector2i(0, -1))
        KEY_S, KEY_DOWN:  RunManager.try_move(Vector2i(0, 1))
        KEY_A, KEY_LEFT:  RunManager.try_move(Vector2i(-1, 0))
        KEY_D, KEY_RIGHT: RunManager.try_move(Vector2i(1, 0))
        KEY_E:            RunManager.try_explore()
        KEY_R:            RunManager.rest()
```

- 일자: `Day 4 / 10`
- 행동: `행동 5 남음`
- 회귀 카운트 / 자원 (기존)
